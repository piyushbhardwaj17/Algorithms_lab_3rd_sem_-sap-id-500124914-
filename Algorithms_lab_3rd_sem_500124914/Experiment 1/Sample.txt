#include <stdio.h>

int binarySearch(int arr[],int beg, int end, int num) {
    if (beg <= end) 
    {
        int mid = (beg + end)/2;
        if (arr[mid]==num)
            return mid;
        if (arr[mid]<num)
            return binarySearch(arr,mid+1,end,num);
        return binarySearch(arr, beg,mid-1,num); }
    return -1;
}
int main() 
{
    int size, num, result;
    printf("Enter the size of the array: ");
    scanf("%d",&size);
    int arr[size];
    printf("Enter %d elements in sorted order\n",size);
    for (int i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Enter the number to search: ");
    scanf("%d", &num);
    result = binarySearch(arr,0, size-1, num);
    if (result !=-1)
        printf("Element found at index %d\n",result);
    else
        printf("Element not found");
    return 0;}

void binary_search(int arr[10], int key, int i, int j){

    if(i == j){

        printf("Not found \n");
        return;
    }

    int mid = (i + j)/2;

    if(arr[mid] == key){

        printf("Element found at %d index \n", i);
        return;
    }

    else if(arr[mid] < key)
        i  = mid + 1;

    else 
        j = mid - 1;

    binary_search(arr, key, i, j);        
}
